Get age from the user
if age is less than or equal to 1, set ageCategory to 'Infant'
else if age is greater than 1 and less than 13, set ageCategory to 'Child'
else if age is greater than or equal to 13 and less than 20, set ageCategory to 'Teenager'
else age is greater than or equal to 20, set ageCategory to 'Adult'
Display ageCategory
